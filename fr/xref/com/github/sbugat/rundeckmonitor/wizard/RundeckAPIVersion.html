5   /**
6    * Possible API version usable with rundeck monitor.
7    * 
8    * @author Sylvain Bugat
9    * 
10   */
11  public enum RundeckAPIVersion {
12  
13  	/** API version 5. */
14  	RUNDECK_APIVERSION_5(Version.V5, "1.4.4"), //$NON-NLS-1$
15  	/** API version 6. */
16  	RUNDECK_APIVERSION_6(Version.V6, "1.5.1"), //$NON-NLS-1$
17  	/** API version 7. */
18  	RUNDECK_APIVERSION_7(Version.V7, "1.5.3"), //$NON-NLS-1$
19  	/** API version 8. */
20  	RUNDECK_APIVERSION_8(Version.V8, "1.6.0"), //$NON-NLS-1$
21  	/** API version 9. */
22  	RUNDECK_APIVERSION_9(Version.V9, "1.6.1"), //$NON-NLS-1$
23  	/** API version 10. */
24  	RUNDECK_APIVERSION_10(Version.V10, "2.0.0"), //$NON-NLS-1$
25  	/** API version 11. */
26  	RUNDECK_APIVERSION_11(Version.V11, "2.1.0"), //$NON-NLS-1$
27  	/** API version 12. */
28  	RUNDECK_APIVERSION_12(Version.V12, "2.2.0"), //$NON-NLS-1$
29  	/** API version 13. */
30  	RUNDECK_APIVERSION_13(Version.V13, "2.5.0"); //$NON-NLS-1$
31  
32  	/** Rundeck API Version. */
33  	private final Version version;
34  	/** First version using this API version. */
35  	private final String sinceRundeckVersion;
36  
37  	/**
38  	 * Copy arguments tp constants.
39  	 * 
40  	 * @param versionArg API version
41  	 * @param sinceRundeckVersionArg first RunDeck version using this API version
42  	 */
43  	private RundeckAPIVersion(final Version versionArg, final String sinceRundeckVersionArg) {
44  
45  		version = versionArg;
46  		sinceRundeckVersion = sinceRundeckVersionArg;
47  	}
48  
49  	/**
50  	 * Get the API version.
51  	 * 
52  	 * @return API version
53  	 */
54  	public Version getVersion() {
55  		return version;
56  	}
57  
58  	/**
59  	 * Get the first RunDeck version using this API version.
60  	 * 
61  	 * @return RunDeck version
62  	 */
63  	public String getSinceReturnVersion() {
64  		return sinceRundeckVersion;
65  	}
66  
67  	/**
68  	 * Return the version and the first API version.
69  	 * 
70  	 * @return displayed String
71  	 */
72  	@Override
73  	public String toString() {
74  		return version + " - since " + sinceRundeckVersion; //$NON-NLS-1$
75  	}
76  }
