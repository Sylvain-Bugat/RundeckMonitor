111 	/**
112 	 * Load configuration
113 	 *
114 	 * @throws IOException in case of loading configuration error
115 	 */
116 	public void loadConfigurationPropertieFile() throws IOException {
117 
118 		//Configuration loading
119 		final Path propertyFile = Paths.get( RUNDECK_MONITOR_PROPERTIES_FILE );
120 		if( ! Files.exists( propertyFile ) ){
121 
122 			throw new FileNotFoundException( RUNDECK_MONITOR_PROPERTIES_FILE );
123 		}
124 
125 		//Load the configuration file and extract properties
126 		final Properties properties = new Properties();
127 		try( final Reader propertyFileReader = Files.newBufferedReader( propertyFile, StandardCharsets.UTF_8 ) ) {
128 			properties.load( propertyFileReader );
129 		}
130 
131 		rundeckUrl = properties.getProperty( RUNDECK_MONITOR_PROPERTY_URL );
132 		rundeckProject = properties.getProperty( RUNDECK_MONITOR_PROPERTY_PROJECT );
133 
134 		rundeckAPIKey = properties.getProperty( RUNDECK_MONITOR_PROPERTY_API_KEY );
135 
136 		rundeckLogin = properties.getProperty( RUNDECK_MONITOR_PROPERTY_LOGIN );
137 		rundeckPassword = properties.getProperty( RUNDECK_MONITOR_PROPERTY_PASSWORD );
138 
139 		rundeckMonitorName = properties.getProperty( RUNDECK_MONITOR_PROPERTY_NAME, RUNDECK_MONITOR_PROPERTY_NAME_DEFAULT_VALUE );
140 		refreshDelay = getIntegerProperty( properties, RUNDECK_MONITOR_PROPERTY_REFRESH_DELAY, RUNDECK_MONITOR_PROPERTY_REFRESH_DELAY_DEFAULT_VALUE );
141 		lateThreshold = getIntegerProperty( properties, RUNDECK_MONITOR_PROPERTY_EXECUTION_LATE_THRESHOLD, RUNDECK_MONITOR_PROPERTY_EXECUTION_LATE_THRESHOLD_DEFAULT_VALUE );
142 		failedJobNumber = getIntegerProperty( properties, RUNDECK_MONITOR_PROPERTY_FAILED_JOB_NUMBER, RUNDECK_MONITOR_PROPERTY_FAILED_JOB_NUMBER_DEFAULT_VALUE );
143 		dateFormat = properties.getProperty( RUNDECK_MONITOR_PROPERTY_DATE_FORMAT, RUNDECK_MONITOR_PROPERTY_DATE_FORMAT_DEFAULT_VALUE );
144 		rundeckAPIversion = getIntegerProperty( properties, RUNDECK_MONITOR_PROPERTY_API_VERSION, RUNDECK_MONITOR_PROPERTY_API_VERSION_DEFAULT_VALUE );
145 		jobTabRedirection = properties.getProperty( RUNDECK_MONITOR_PROPERTY_FAILED_JOB_REDIRECTION, RUNDECK_MONITOR_PROPERTY_FAILED_JOB_REDIRECTION_DEFAULT_VALUE );
146 		versionCheckerDisabled = getBooleanProperty( properties, RUNDECK_MONITOR_PROPERTY_DISABLE_VERSION_CHECKER, RUNDECK_MONITOR_PROPERTY_DISABLE_VERSION_CHECKER_DEFAULT_VALUE );
147 		interfaceType = properties.getProperty( RUNDECK_MONITOR_PROPERTY_INTERFACE_TYPE, RUNDECK_MONITOR_PROPERTY_INTERFACE_TYPE_DEFAULT_VALUE );
148 	}

150 	/**
151 	 * Check configuration
152 	 *
153 	 * @throws InvalidPropertyException
154 	 * @throws MissingPropertyException
155 	 */
156 	public void verifyConfiguration() throws MissingPropertyException, InvalidPropertyException  {
157 
158 		//Configuration checking
159 
160 		checkMandatoryStringProperty( rundeckUrl, RUNDECK_MONITOR_PROPERTY_URL );
161 		checkMandatoryStringProperty( rundeckProject, RUNDECK_MONITOR_PROPERTY_PROJECT );
162 
163 		boolean missingAPIKey = false;
164 		try {
165 			checkMandatoryStringProperty( rundeckAPIKey, RUNDECK_MONITOR_PROPERTY_API_KEY );
166 		}
167 		catch( final MissingPropertyException | InvalidPropertyException e ) {
168 			missingAPIKey = false;
169 		}
170 
171 		if( missingAPIKey ) {
172 			checkMandatoryStringProperty( rundeckLogin, RUNDECK_MONITOR_PROPERTY_LOGIN );
173 			checkMandatoryStringProperty( rundeckPassword, RUNDECK_MONITOR_PROPERTY_PASSWORD );
174 		}
175 
176 		//Test the configured date format
177 		try {
178 			new SimpleDateFormat( dateFormat );
179 		}
180 		catch( final IllegalArgumentException e ) {
181 			dateFormat = RUNDECK_MONITOR_PROPERTY_DATE_FORMAT_DEFAULT_VALUE;
182 		}
183 
184 		//Test the configured tab redirection
185 		try {
186 			JobTabRedirection.valueOf( jobTabRedirection );
187 		}
188 		catch( final IllegalArgumentException e ) {
189 			jobTabRedirection = RUNDECK_MONITOR_PROPERTY_FAILED_JOB_REDIRECTION_DEFAULT_VALUE;
190 		}
191 
192 	}

194 	private static void checkMandatoryStringProperty( final String property, final String propertyName ) throws MissingPropertyException, InvalidPropertyException {
195 
196 		if( null == property ) {
197 			throw new MissingPropertyException( propertyName );
198 		}
199 		else if( property.isEmpty() ) {
200 			throw new InvalidPropertyException( propertyName, property );
201 		}
202 	}

204 	private static int getIntegerProperty( final Properties properties, final String propertyName, final int defaultValue ) {
205 
206 		final String propertyValue = properties.getProperty( propertyName, String.valueOf( defaultValue ) );
207 
208 		if( propertyValue.isEmpty() ) {
209 			return defaultValue;
210 		}
211 
212 		try {
213 			return Integer.parseInt( propertyValue );
214 		}
215 		catch( final NumberFormatException e ) {
216 			return defaultValue;
217 		}
218 	}
219 
220 	private static boolean getBooleanProperty( final Properties properties, final String propertyName, final boolean defaultValue ) {
221 
222 		return Boolean.parseBoolean( properties.getProperty( propertyName, String.valueOf( defaultValue ) ) );
223 	}

225 	public void saveMonitorConfigurationPropertieFile() throws IOException {
226 
227 		//Add all properties
228 		final Properties properties = new Properties();
229 
230 		properties.put( RUNDECK_MONITOR_PROPERTY_URL, rundeckUrl );
231 		properties.put( RUNDECK_MONITOR_PROPERTY_API_KEY, rundeckAPIKey );
232 		properties.put( RUNDECK_MONITOR_PROPERTY_LOGIN, rundeckLogin );
233 		properties.put( RUNDECK_MONITOR_PROPERTY_PASSWORD, rundeckPassword );
234 		properties.put( RUNDECK_MONITOR_PROPERTY_PROJECT, rundeckProject );
235 		properties.put( RUNDECK_MONITOR_PROPERTY_NAME, rundeckMonitorName);
236 		properties.put( RUNDECK_MONITOR_PROPERTY_REFRESH_DELAY, String.valueOf( refreshDelay ) );
237 		properties.put( RUNDECK_MONITOR_PROPERTY_EXECUTION_LATE_THRESHOLD, String.valueOf( lateThreshold ) );
238 		properties.put( RUNDECK_MONITOR_PROPERTY_FAILED_JOB_NUMBER, String.valueOf( failedJobNumber ) );
239 		properties.put( RUNDECK_MONITOR_PROPERTY_DATE_FORMAT, dateFormat );
240 		properties.put( RUNDECK_MONITOR_PROPERTY_API_VERSION, String.valueOf( rundeckAPIversion ) );
241 		properties.put( RUNDECK_MONITOR_PROPERTY_FAILED_JOB_REDIRECTION, jobTabRedirection );
242 		properties.put( RUNDECK_MONITOR_PROPERTY_DISABLE_VERSION_CHECKER, String.valueOf( versionCheckerDisabled ) );
243 		properties.put( RUNDECK_MONITOR_PROPERTY_INTERFACE_TYPE, interfaceType );
244 
245 		//Comment header
246 		final StringBuilder commentStringBuilder = new StringBuilder();
247 		commentStringBuilder.append( "Generated by RundeckMonitor wizard at: " ); //$NON-NLS-1$
248 		try {
249 			final SimpleDateFormat formatter = new SimpleDateFormat( dateFormat );
250 			commentStringBuilder.append( formatter.format( new Date() ) );
251 		}
252 		catch( final IllegalArgumentException e ) {
253 			commentStringBuilder.append( new Date() );
254 		}
255 		commentStringBuilder.append(  System.lineSeparator() );
256 		commentStringBuilder.append( "-------------------------------------------------------" ); //$NON-NLS-1$
257 
258 		//Properties file writing
259 		final Path propertyFile = Paths.get( RUNDECK_MONITOR_PROPERTIES_FILE );
260 		try( final Writer propertyFilewriter = Files.newBufferedWriter( propertyFile, StandardCharsets.UTF_8 ) ) {
261 			properties.store( propertyFilewriter, commentStringBuilder.toString() );
262 		}
263 	}

276 	public static boolean propertiesFileUpdated( final Date date ) {
277 
278 		final Path propertyFile = Paths.get( RUNDECK_MONITOR_PROPERTIES_FILE );
279 		if( Files.exists( propertyFile ) ) {
280 
281 			Date fileTime;
282 			try {
283 				fileTime = new Date( Files.getLastModifiedTime( propertyFile ).toMillis() );
284 			}
285 			catch( final IOException e ) {
286 				return false;
287 			}
288 
289 			return fileTime.after( date );
290 		}
291 
292 		return false;
293 	}
