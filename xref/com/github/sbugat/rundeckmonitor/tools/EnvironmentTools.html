3   /**
4    * Tools class to detect Operating System type
5    *
6    * @author Sylvain Bugat
7    *
8    */
9   public class EnvironmentTools {
10  
11  	private static final String OS_NAME_PROPERTY = "os.name"; //$NON-NLS-1$
12  	private static final String WINDOWS_OS_NAME = "windows"; //$NON-NLS-1$
13  
14  	/**
15  	 * Check if the operating system is a windows based on a property, otherwise it's a Linux/Mac-OS
16  	 *
17  	 * @return true if the OS is a windows*
18  	 */
19  	public static boolean isWindows() {
20  
21  		final String operatingSystem = System.getProperty( OS_NAME_PROPERTY );
22  
23  		if( null == operatingSystem ) {
24  			return false;
25  		}
26  
27  		return operatingSystem.toLowerCase().startsWith( WINDOWS_OS_NAME );
28  	}
29  }
