1   package com.github.sbugat.rundeckmonitor.tools;
2   
3   import java.util.Locale;
4   
5   import org.slf4j.ext.XLogger;
6   import org.slf4j.ext.XLoggerFactory;
7   
8   /**
9    * Tools class to detect Operating System type.
10   * 
11   * @author Sylvain Bugat
12   * 
13   */
14  public final class EnvironmentTools {
15  
16  	/** SLF4J XLogger. */
17  	private static final XLogger LOG = XLoggerFactory.getXLogger(EnvironmentTools.class);
18  
19  	/** Windows OS name property. */
20  	private static final String OS_NAME_PROPERTY = "os.name"; //$NON-NLS-1$
21  	/** Windows name in OS name property. */
22  	private static final String WINDOWS_OS_NAME = "windows"; //$NON-NLS-1$
23  
24  	/**
25  	 * Private constructor to prevent Instantiating.
26  	 */
27  	private EnvironmentTools() {
28  		throw new UnsupportedOperationException();
29  	}
30  
31  	/**
32  	 * Check if the operating system is a windows based on a property, otherwise it's a Linux/Mac-OS.
33  	 * 
34  	 * @return true if the OS is a windows*
35  	 */
36  	public static boolean isWindows() {
37  
38  		LOG.entry();
39  		final String operatingSystem = System.getProperty(OS_NAME_PROPERTY);
40  
41  		if (null == operatingSystem) {
42  			LOG.exit(false);
43  			return false;
44  		}
45  
46  		final boolean isWindows = operatingSystem.toLowerCase(Locale.getDefault()).startsWith(WINDOWS_OS_NAME);
47  		LOG.exit(isWindows);
48  		return isWindows;
49  	}
50  }
