1   package com.github.sbugat.rundeckmonitor.tools;
2   
3   import org.slf4j.ext.XLogger;
4   import org.slf4j.ext.XLoggerFactory;
5   
6   /**
7    * Tools class to do System class calls.
8    * 
9    * @author Sylvain Bugat
10   * 
11   */
12  public final class SystemTools {
13  
14  	/** SLF4J XLogger. */
15  	private static final XLogger LOG = XLoggerFactory.getXLogger(SystemTools.class);
16  
17  	/**
18  	 * Normal return code.
19  	 */
20  	public static final int EXIT_CODE_OK = 0;
21  
22  	/**
23  	 * Error return code.
24  	 */
25  	public static final int EXIT_CODE_ERROR = 1;
26  
27  	/**
28  	 * Unsupported tray icon by the system return code.
29  	 */
30  	public static final int EXIT_CODE_TRAY_ICON_UNSUPPORTED = 2;
31  
32  	/**
33  	 * tray icon initialization error return code.
34  	 */
35  	public static final int EXIT_CODE_TRAY_ICON_ERROR = 3;
36  
37  	/**
38  	 * Private constructor to prevent Instantiating.
39  	 */
40  	private SystemTools() {
41  		throw new UnsupportedOperationException();
42  	}
43  
44  	/**
45  	 * Log and call System.exit method.
46  	 * 
47  	 * @param returnCode return code of the java program, argument to System.exit( returnCode )
48  	 */
49  	public static void exit(final int returnCode) {
50  
51  		LOG.entry(returnCode);
52  
53  		if (EXIT_CODE_OK == returnCode) {
54  			LOG.info("RunDeck Monitor exit with code {}", EXIT_CODE_OK); //$NON-NLS-1$
55  		}
56  		else {
57  			LOG.error("RunDeck Monitor exit with code {}", returnCode); //$NON-NLS-1$
58  		}
59  
60  		// Stop the JVM with the expected return code
61  		LOG.exit(returnCode);
62  		System.exit(returnCode);
63  	}
64  }
