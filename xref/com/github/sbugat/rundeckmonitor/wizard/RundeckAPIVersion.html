25  	private final String sinceRundeckVersion;

29  		version = versionArg;
30  		sinceRundeckVersion = sinceRundeckVersionArg;
31  	}
32  
33  	public Version getVersion() {

35  	}
36

39  	}
40  
41  	public String toString() {
