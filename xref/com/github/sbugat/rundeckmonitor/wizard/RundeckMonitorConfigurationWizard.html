21  /**
22   * Class of the configuration Wizard
23   *
24   * @author Sylvain Bugat
25   *
26   */
27  public class RundeckMonitorConfigurationWizard {
28  
29  	private static final String WIZARD_FRAME_TITLE = "RundeckMonitor configuration wizard"; //$NON-NLS-1$
30  
31  	private static final String BACK_BUTTON_LABEL = "Back"; //$NON-NLS-1$
32  	private static final String CANCEL_BUTTON_LABEL = "Cancel"; //$NON-NLS-1$
33  	private static final String FINISH_BUTTON_LABEL = "Finish"; //$NON-NLS-1$
34  	private static final String NEXT_BUTTON_LABEL = "Next"; //$NON-NLS-1$
35  
36  	private Map<ConfigurationWizardStep, WizardPanelDescriptor> map = new HashMap<>();
37  	private ConfigurationWizardStep currentStep;
38  
39  	private final JFrame wizardFrame;
40  
41  	private JPanel cardPanel;
42  	private CardLayout cardLayout;
43  
44  	/**Back/previous step button of the wizard*/
45  	private JButton backButton;
46  	/**Next step button of the wizard*/
47  	private JButton nextButton;
48  
49  	public RundeckMonitorConfigurationWizard( final RundeckMonitorConfiguration rundeckMonitorConfiguration, final boolean exitOnClose ) {
50  
51  		wizardFrame = new JFrame();
52  		wizardFrame.setTitle( WIZARD_FRAME_TITLE );
53  
54  		if( exitOnClose ) {
55  			wizardFrame.setDefaultCloseOperation( JFrame.EXIT_ON_CLOSE );
56  		}
57  		else {
58  			wizardFrame.setDefaultCloseOperation( JFrame.DISPOSE_ON_CLOSE );
59  		}
60  
61  		//Initialize displayed components
62  		initComponents( exitOnClose );
63  
64  		//Initialize steps and set the first step
65  		WizardPanelDescriptor wpd1 = new RundeckConfigurationWizardPanelDescriptor( ConfigurationWizardStep.RUNDECK_STEP, null, ConfigurationWizardStep.PROJECT_STEP, rundeckMonitorConfiguration );
66  		WizardPanelDescriptor wpd2 = new ProjectConfigurationWizardPanelDescriptor( ConfigurationWizardStep.PROJECT_STEP, ConfigurationWizardStep.RUNDECK_STEP, ConfigurationWizardStep.MONITOR_STEP, rundeckMonitorConfiguration );
67  		WizardPanelDescriptor wpd3 = new MonitorConfigurationWizardPanelDescriptor( ConfigurationWizardStep.MONITOR_STEP, ConfigurationWizardStep.PROJECT_STEP, null, rundeckMonitorConfiguration );
68  
69  		registerWizardPanel( wpd1 );
70  		registerWizardPanel( wpd2 );
71  		registerWizardPanel( wpd3 );
72  		setCurrentPanel( ConfigurationWizardStep.RUNDECK_STEP, false );
73  
74  		//Resize the frame
75  		wizardFrame.pack();
76  		wizardFrame.setResizable( false );
77  		//Center the frame on the screen
78  		wizardFrame.setLocationRelativeTo( null );
79  
80  		//Display the frame
81  		wizardFrame.setVisible(true);
82  	}
83  
84  	private void initComponents( final boolean exitOnClose ) {
85  
86  		final JPanel buttonPanel = new JPanel();
87  		final Box buttonBox = new Box( BoxLayout.X_AXIS );
88  
89  		cardPanel = new JPanel();
90  		cardPanel.setBorder( new EmptyBorder( new Insets( 5, 10, 5, 10 ) ) );
91  
92  		cardLayout = new CardLayout();
93  		cardPanel.setLayout( cardLayout );
94  		backButton = new JButton( BACK_BUTTON_LABEL );
95  		nextButton = new JButton( NEXT_BUTTON_LABEL );
96  		final JButton cancelButton = new JButton( CANCEL_BUTTON_LABEL );
97  
98  		final ActionListener backListener = new ActionListener() {
99  			@SuppressWarnings("synthetic-access")
100 			public void actionPerformed( final ActionEvent e) {
101 				final ConfigurationWizardStep dest = map.get( currentStep ).getBack();
102 				setCurrentPanel( dest, false );
103 			}
104 		};
105 		final ActionListener nextListener = new ActionListener() {
106 			@SuppressWarnings("synthetic-access")
107 			public void actionPerformed( final ActionEvent e) {
108 				final ConfigurationWizardStep dest = map.get( currentStep ).getNext();
109 
110 				if( null == dest ) {
111 
112 					final WizardPanelDescriptor oldPanelDescriptor = map.get( currentStep );
113 
114 					//Write configuration
115 					oldPanelDescriptor.validate();
116 
117 					wizardFrame.setVisible( false );
118 					wizardFrame.dispose();
119 				}
120 				else {
121 					setCurrentPanel( dest, true );
122 				}
123 			}
124 		};
125 		final ActionListener cancelListener = new ActionListener() {
126 			@SuppressWarnings("synthetic-access")
127 			public void actionPerformed( final ActionEvent e) {
128 
129 				if( exitOnClose ) {
130 					System.exit( 0 );
131 				}
132 				else {
133 					wizardFrame.setVisible( false );
134 					wizardFrame.dispose();
135 				}
136 			}
137 		};
138 
139 		backButton.addActionListener( backListener );
140 		nextButton.addActionListener( nextListener );
141 		cancelButton.addActionListener( cancelListener );
142 
143 		buttonPanel.setLayout( new BorderLayout() );
144 		buttonPanel.add( new JSeparator(), BorderLayout.NORTH );
145 
146 		buttonBox.setBorder( new EmptyBorder( new Insets( 5, 10, 5, 10 ) ) );
147 		buttonBox.add( backButton );
148 		buttonBox.add( Box.createHorizontalStrut( 10 ) );
149 		buttonBox.add( nextButton );
150 		buttonBox.add( Box.createHorizontalStrut( 30 ) );
151 		buttonBox.add( cancelButton );
152 		buttonPanel.add( buttonBox, BorderLayout.EAST );
153 		wizardFrame.getContentPane().add( buttonPanel, BorderLayout.SOUTH );
154 		wizardFrame.getContentPane().add( cardPanel, BorderLayout.CENTER );
155 	}
156 
157 	public void registerWizardPanel( final WizardPanelDescriptor panel ) {
158 		cardPanel.add( panel.getPanelComponent(), panel.getPanelDescriptorIdentifier().toString() );
159 		map.put( panel.getPanelDescriptorIdentifier(), panel );
160 	}
161 
162 	public void setCurrentPanel( final ConfigurationWizardStep id, final boolean next ) {
163 
164 		//If going to the next panel, validate it and don't advance if validation failed
165 		if( next && currentStep != null && ! map.get( currentStep ).validate()) {
166 			return;
167 		}
168 
169 		currentStep = id;
170 
171 		map.get( id ).aboutToDisplayPanel();
172 
173 		if( null == map.get( id ).getNext() ) {
174 			nextButton.setText( FINISH_BUTTON_LABEL );
175 		}
176 		else {
177 			nextButton.setText( NEXT_BUTTON_LABEL );
178 		}
179 
180 		backButton.setVisible( null != map.get( id ).getBack() );
181 
182 		cardLayout.show( cardPanel, id.toString() );
183 	}
