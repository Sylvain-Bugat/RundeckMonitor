24  /**
25   * Simple generic version checker on GitHub, inpect target jar and local jar build date to determinated if an update is available
26   * If one is found, download the full jar and replace the original jar via a double restart
27   *
28   * The checker use an independant thread to check and download the file.
29   * The main thread have to check is the download is done and order to restart the program
30   *
31   * @author Sylvain Bugat
32   *
33   */
34  public class VersionChecker implements Runnable{
35  
36  	private static final String JAR_EXTENSION = ".jar"; //$NON-NLS-1$
37  	private static final String TMP_EXTENSION = ".tmp"; //$NON-NLS-1$
38  	private static final String WINDOWS_EXE_EXTENSION = ".exe"; //$NON-NLS-1$
39  	private static final String JAVA_HOME_PROPERTY = "java.home"; //$NON-NLS-1$
40  	private static final String TARGET_DIRECTORY = "target"; //$NON-NLS-1$
41  
42  	private static final String BIN_DIRECTORY_AND_JAVA = "bin" + FileSystems.getDefault().getSeparator() + "java"; //$NON-NLS-1$ //$NON-NLS-2$
43  
44  	private static final String JAR_ARGUMENT = "-jar"; //$NON-NLS-1$
45  
46  	/**Root URL of the GitHub project to update*/
47  	private final String gitHubUser;
48  	private final String gitHubRepository;
49  
50  	/**Maven artifact identifier*/
51  	private final String mavenArtifactId;
52  
53  	private final String jarWithDependenciesSuffix;
54  
55  	/**Indicate if the download is completed*/
56  	private boolean downloadDone;
57  
58  	/**Indicate if the version checker is disabled*/
59  	private boolean versionCheckerDisabled;
60  
61  	private String downloadedJar;
62  
63  	/**
64  	 * Initialize the version checker with jar artifact and suffixnames and path to GitHub
65  	 *
66  	 * @param gitHubUserArg
67  	 * @param gitHubRepositoryArg
68  	 * @param mavenArtifactIdArg
69  	 * @param jarWithDependenciesSuffixArg
70  	 */
71  	public VersionChecker( final String gitHubUserArg, final String gitHubRepositoryArg, final String mavenArtifactIdArg, final String jarWithDependenciesSuffixArg ) {
72  
73  		gitHubUser = gitHubUserArg;
74  		gitHubRepository = gitHubRepositoryArg;
75  
76  		mavenArtifactId = mavenArtifactIdArg;
77  		jarWithDependenciesSuffix = jarWithDependenciesSuffixArg;
78  	}
79  
80  	/**
81  	 * Background thread launched to check the version on GitHub
82  	 */
83  	@Override
84  	public void run() {
85  
86  		final String currentJar = currentJar();
87  
88  		if( null == currentJar ) {
89  			return;
90  		}
91  
92  		try {
93  
94  			final GitHubClient gitHubClient = new GitHubClient();
95  
96  			final RepositoryService rs = new RepositoryService( gitHubClient );
97  			final Repository repository = rs.getRepository( gitHubUser, gitHubRepository );
98  
99  			final String currentVersion = 'v' + currentJar.replaceFirst( "^" + mavenArtifactId + '-', "" ).replaceFirst( jarWithDependenciesSuffix + ".*$", "" ); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$ //$NON-NLS-4$
100 			RepositoryTag recentRelease = null;
101 			for( final RepositoryTag tag : rs.getTags( repository ) ) {
102 
103 				if( null != recentRelease && tag.getName().compareTo( recentRelease.getName() ) > 0 ) {
104 					recentRelease = tag;
105 				}
106 				else if( ( tag.getName() ).compareTo( currentVersion ) > 0 ) {
107 					recentRelease = tag;
108 				}
109 			}
110 
111 			if( null == recentRelease  ) {
112 				return;
113 			}
114 
115 			if( ! findAndDownloadReleaseJar( recentRelease, true ) ) {
116 				findAndDownloadReleaseJar( recentRelease, false );
117 			}
118 		}
119 		catch( final Exception e) {
120 
121 			//Ignore any error during update process
122 			//Just delete the temporary file
123 			cleanOldAndTemporaryJar();
124 		}
125 	}
126 
127 	private boolean findAndDownloadReleaseJar( final RepositoryTag release, final boolean withDependenciesSuffix ) throws IOException {
128 
129 		final String jarSuffix;
130 		if( withDependenciesSuffix ) {
131 			jarSuffix = jarWithDependenciesSuffix;
132 		}
133 		else {
134 			jarSuffix = ""; //$NON-NLS-1$
135 		}
136 		try( final InputStream remoteJarInputStream = new URL( release.getZipballUrl() ).openStream() ) {
137 
138 			final ZipInputStream zis = new ZipInputStream( remoteJarInputStream );
139 
140 			ZipEntry entry = zis.getNextEntry();
141 
142 			while( null != entry ) {
143 
144 				if( entry.getName().matches( ".*/" + TARGET_DIRECTORY + '/' + mavenArtifactId + "-[0-9\\.]*" + jarSuffix + JAR_EXTENSION  ) ) { //$NON-NLS-1$ //$NON-NLS-2$
145 
146 					final Object[] options = { "Yes", "No", "Never ask me again" }; //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
147 					final int confirmDialogChoice = JOptionPane.showOptionDialog( null, "An update is available, download it? (" + entry.getCompressedSize() / 1_048_576 + "MB)", "Rundeck Monitor update found!", JOptionPane.YES_NO_CANCEL_OPTION, JOptionPane.QUESTION_MESSAGE, null, options, options[ 0 ] );  //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
148 					if( JOptionPane.YES_OPTION == confirmDialogChoice ) {
149 
150 						final String jarFileBaseName = entry.getName().replaceFirst( "^.*/", "" ); //$NON-NLS-1$ //$NON-NLS-2$
151 
152 						downloadFile( zis, jarFileBaseName + TMP_EXTENSION );
153 						Files.move( Paths.get( jarFileBaseName + TMP_EXTENSION ), Paths.get( jarFileBaseName ) );
154 
155 						downloadedJar = jarFileBaseName;
156 						downloadDone = true;
157 					}
158 					else if( JOptionPane.CANCEL_OPTION == confirmDialogChoice ) {
159 
160 						versionCheckerDisabled = true;
161 					}
162 
163 					return true;
164 				}
165 
166 				entry = zis.getNextEntry();
167 			}
168 		}
169 
170 		return false;
171 	}

173 	public boolean restart() {
174 
175 		if( Files.exists( Paths.get( downloadedJar ) ) ) {
176 
177 			try {
178 
179 				final ProcessBuilder processBuilder = new ProcessBuilder( getJavaExecutable(), JAR_ARGUMENT, downloadedJar );
180 				processBuilder.start();
181 
182 				return true;
183 			}
184 			catch( final IOException e ) {
185 
186 				//Ignore any error during restart process
187 			}
188 		}
189 
190 		return false;
191 	}

193 	public void cleanOldAndTemporaryJar() {
194 
195 		String currentJar = currentJar();
196 
197 		try( final DirectoryStream<Path> directoryStream = Files.newDirectoryStream( Paths.get( "." ) ) ) { //$NON-NLS-1$
198 
199 			for( final Path path : directoryStream ) {
200 
201 				final String fileName = path.getFileName().toString();
202 				if( fileName.startsWith( mavenArtifactId) ) {
203 
204 					if( fileName.endsWith( JAR_EXTENSION ) && null != currentJar && currentJar.compareTo( fileName ) > 0 ) {
205 
206 						deleteJar( path );
207 					}
208 					else if( fileName.endsWith( JAR_EXTENSION + TMP_EXTENSION ) ) {
209 
210 						deleteJar( path );
211 					}
212 				}
213 			}
214 		}
215 		catch ( final IOException ex ) {
216 			//Ignore any error during the delete process
217 		}
218 	}

220 	private static void deleteJar( final Path jarFileToDelete ) {
221 
222 		if( Files.exists( jarFileToDelete ) ) {
223 
224 			try {
225 				Files.delete( jarFileToDelete );
226 			}
227 			catch ( final IOException e ) {
228 
229 				//Ignore any error during the delete process
230 			}
231 		}
232 	}

234 	private String currentJar() {
235 
236 		String currentJar = null;
237 		try( final DirectoryStream<Path> directoryStream = Files.newDirectoryStream( Paths.get( "." ) ) ) { //$NON-NLS-1$
238 
239 			for( final Path path : directoryStream ) {
240 
241 				final String fileName = path.getFileName().toString();
242 				if( fileName.startsWith( mavenArtifactId ) && fileName.endsWith( JAR_EXTENSION ) && ( null == currentJar || currentJar.compareTo( fileName ) < 0 ) ) {
243 
244 					currentJar = fileName;
245 				}
246 			}
247 		}
248 		catch ( final IOException e ) {
249 			//Ignore any error during the process
250 			currentJar = null;
251 		}
252 
253 		return currentJar;
254 	}

283 	/**
284 	 * Get the java executable
285 	 *
286 	 * @return absolte path to the java executable
287 	 * @throws NoSuchFileException if the java executable is not found
288 	 */
289 	private static String getJavaExecutable() throws NoSuchFileException {
290 
291 		final String javaDirectory = System.getProperty( JAVA_HOME_PROPERTY );
292 
293 		if ( javaDirectory == null ) {
294 			throw new IllegalStateException( JAVA_HOME_PROPERTY );
295 		}
296 
297 		final String javaExecutableFilePath;
298 		//Add .exe extension on Windows OS
299 		if ( EnvironmentTools.isWindows() ) {
300 			javaExecutableFilePath = javaDirectory + FileSystems.getDefault().getSeparator() + BIN_DIRECTORY_AND_JAVA + WINDOWS_EXE_EXTENSION;
301 		}
302 		else {
303 			javaExecutableFilePath = javaDirectory + FileSystems.getDefault().getSeparator() + BIN_DIRECTORY_AND_JAVA;
304 		}
305 
306 		//Check if the executable exists and is executable
307 		final Path javaExecutablePath = Paths.get( javaExecutableFilePath );
308 		if ( ! Files.exists( javaExecutablePath ) || ! Files.isExecutable( javaExecutablePath ) ) {
309 			throw new NoSuchFileException( javaExecutableFilePath );
310 		}
311 
312 		return javaExecutableFilePath;
313 	}
