51  	private final Image IMAGE_LATE = Toolkit.getDefaultToolkit().getImage( getClass().getClassLoader().getResource( "LATE.png" ) ); //$NON-NLS-1$

59  	private final Image IMAGE_DISCONNECTED = Toolkit.getDefaultToolkit().getImage( getClass().getClassLoader().getResource( "DISCONNECTED.png" ) ); //$NON-NLS-1$

95  	/**
96  	 * Initialize the tray icon for the rundeckMonitor if the OS is compatible with it
97  	 *
98  	 * @param rundeckMonitorConfigurationArg loaded configuration
99  	 * @param rundeckMonitorStateArg state of the rundeck monitor
100 	 */
101 	public RundeckMonitorTrayIcon( final RundeckMonitorConfiguration rundeckMonitorConfigurationArg, final RundeckMonitorState rundeckMonitorStateArg ) {
102 
103 		rundeckMonitorConfiguration = rundeckMonitorConfigurationArg;
104 		rundeckMonitorState = rundeckMonitorStateArg;
105 
106 		if( SystemTray.isSupported() ) {
107 
108 			//Try to use the system Look&Feel
109 			try {
110 				UIManager.setLookAndFeel( UIManager.getSystemLookAndFeelClassName() );
111 			}
112 			catch( final ClassNotFoundException | InstantiationException | IllegalAccessException | UnsupportedLookAndFeelException e ) {
113 
114 				//If System Look&Feel is not supported, stay with the default one
115 			}
116 
117 			// Get the system default browser to open execution details
118 			desktop = Desktop.getDesktop();
119 
120 			//Edit configuration listener
121 			configurationListener = new ActionListener() {
122 				@SuppressWarnings("synthetic-access")
123 				public void actionPerformed( final ActionEvent e) {
124 					new RundeckMonitorConfigurationWizard( new RundeckMonitorConfiguration( rundeckMonitorConfiguration ), false );
125 				}
126 			};
127 
128 			//Rundeck monitor about
129 			aboutListener = new ActionListener() {
130 				@SuppressWarnings("synthetic-access")
131 				public void actionPerformed( final ActionEvent e) {
132 
133 					try {
134 						final URI executionURI = new URI( RUNDECK_MONITOR_PROJECT_URL );
135 						desktop.browse( executionURI );
136 					}
137 					catch ( final URISyntaxException | IOException exception) {
138 
139 						final StringWriter stringWriter = new StringWriter();
140 						exception.printStackTrace( new PrintWriter( stringWriter ) );
141 						JOptionPane.showMessageDialog( null, exception.getMessage() + System.lineSeparator() + stringWriter.toString(), "RundeckMonitor redirection error", JOptionPane.ERROR_MESSAGE ); //$NON-NLS-1$
142 					}
143 				}
144 			};
145 
146 			//Get the system tray
147 			tray = SystemTray.getSystemTray();
148 
149 			//Rundeck monitor exit
150 			exitListener = new ActionListener() {
151 				@SuppressWarnings("synthetic-access")
152 				public void actionPerformed( final ActionEvent e) {
153 					tray.remove( trayIcon );
154 					System.exit( 0 );
155 				}
156 			};
157 
158 			hiddenDialog = new JDialog ();
159 			hiddenDialog.setSize( 10, 10 );
160 
161 			hiddenDialog.addWindowFocusListener(new WindowFocusListener () {
162 
163 				@Override
164 				public void windowLostFocus ( final WindowEvent e ) {
165 					hiddenDialog.setVisible( false );
166 				}
167 
168 				@Override
169 				public void windowGainedFocus ( final WindowEvent e ) {
170 					//Nothing to do
171 				}
172 			});
173 		}
174 		else {
175 			//if the System is not compatible with SystemTray
176 			tray = null;
177 			desktop = null;
178 			configurationListener = null;
179 			aboutListener = null;
180 			exitListener= null;
181 
182 			JOptionPane.showMessageDialog( null, "SystemTray cannot be initialized", "RundeckMonitor initialization error", JOptionPane.ERROR_MESSAGE ); //$NON-NLS-1$ //$NON-NLS-2$
183 
184 			System.exit( 2 );
185 		}
186 	}

195 	/**
196 	 * Update the image of the tray icon
197 	 */
198 	public void updateTrayIcon() {
199 
200 		if( rundeckMonitorState.isDisconnected() ) {
201 			trayIcon.setImage( IMAGE_DISCONNECTED );
202 		}
203 		else if( rundeckMonitorState.isFailedJobs() ) {
204 
205 			if( rundeckMonitorState.isLateJobs() ) {
206 				trayIcon.setImage( IMAGE_KO_LATE );
207 			}
208 			else {
209 				trayIcon.setImage( IMAGE_KO );
210 			}
211 		}
212 		else if( rundeckMonitorState.isLateJobs() ) {
213 			trayIcon.setImage( IMAGE_LATE );
214 		}
215 		else {
216 			trayIcon.setImage( IMAGE_OK );
217 		}
218 	}
219 
220 	public void reloadConfiguration() {
221 
222 		newLateProcess.clear();
223 		newFailedProcess.clear();
224 
225 		trayIcon.setToolTip( rundeckMonitorConfiguration.getRundeckMonitorName() );
226 	}

229 	 * remove the RundeckMonitor icon from the system tray
230 	 */

232 
233 		tray.remove( trayIcon );
234 	}
